```javascript
// 테마별 종목 필터링
function getStocksByTheme(themeName) {
  const theme = themes.find(t => t.theme_name === themeName);
  return theme ? theme.stocks : [];
}

// 종목 검색 후 테마 표시
function findThemes(stockCode) {
  return themes
    .filter(theme => 
      theme.stocks.some(s => s.code === stockCode)
    )
    .map(theme => theme.theme_name);
}
```
